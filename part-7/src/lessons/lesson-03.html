JavaScript with ScrollTrigger
// Register ScrollTrigger plugin
gsap.registerPlugin(ScrollTrigger);

// Split the hero text into characters using custom splitText
const split = splitText("#hero-text", "chars");

// Set initial state - hidden below
gsap.set(split, { opacity: 0, y: 100 });

// Hero text animation - triggered on scroll
gsap.to(split, {
  opacity: 1,
  y: 0,
  stagger: 0.05,
  duration: 1,
  ease: "power3.out",
  scrollTrigger: {
    trigger: "#hero-text",
    start: "top 80%",
    end: "top 30%",
    scrub: 1 // Smooth scrubbing
  }
});

// Content fade in on scroll
gsap.to("#scroll-content", {
  opacity: 1,
  y: 0,
  duration: 1,
  ease: "power2.out",
  scrollTrigger: {
    trigger: "#scroll-content",
    start: "top 80%",
    toggleActions: "play none none reverse"
  }
});
Key Concepts
// ScrollTrigger options:
// start: "top 80%" - element top at 80% from viewport top
// end: "top 30%" - animation completes at 30%
// scrub: 1 - animation tied to scroll position
// toggleActions - what happens on enter/leave
//   "play none none reverse"
