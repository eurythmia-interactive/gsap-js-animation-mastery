JavaScript Animation
// Custom splitText function (free alternative)
const chars = splitText(heroElement, "chars");

// Set initial state - characters start below and invisible
gsap.set(chars, {
  opacity: 0,
  y: 100,
  rotationX: -90
});

// Create animation timeline
const tl = gsap.timeline();

tl.to(chars, {
  opacity: 1,
  y: 0,
  rotationX: 0,
  stagger: 0.03,
  duration: 0.8,
  ease: "back.out(1.7)"
})
.to("#hero-subtitle", {
  opacity: 1,
  y: 0,
  duration: 0.6,
  ease: "power2.out"
}, "-=0.3");

// Control functions
function playAnimation() { tl.play(); }
function resetAnimation() {
  gsap.set(chars, { opacity: 0, y: 100, rotationX: -90 });
  tl.restart();
}
Key Concepts
// Custom splitText creates spans for each character
  splitText(element, "chars")  // Returns array of spans
  splitText(element, "words")  // Returns array of word spans

// stagger: 0.03 means 30ms between each character
// back.out(1.7) creates a bouncy overshoot effect
// rotationX gives a 3D flip feeling
