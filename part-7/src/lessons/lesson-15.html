JavaScript Animation
// Get timeline items
const items = gsap.utils.toArray(".timeline-item");
const dots = gsap.utils.toArray(".timeline-dot");

// Set initial state
gsap.set(items, { opacity: 0, x: -20 });
gsap.set("#timeline-progress", { height: 0 });

// Create timeline
const tl = gsap.timeline();

tl
  // Progress line draws down
  .to("#timeline-progress", {
    height: "100%",
    duration: 2,
    ease: "power2.inOut"
  })
  // Items slide in with stagger
  .to(items, {
    opacity: 1,
    x: 0,
    stagger: 0.4,
    duration: 0.6,
    ease: "power2.out"
  }, "-=1.5")  // Overlap with progress
  // Dots activate one by one
  .to(dots, {
    background: "#3b82f6",
    borderColor: "#3b82f6",
    stagger: 0.4,
    duration: 0.3,
    ease: "power2.out"
  }, "-=2");
Key Concepts
// Progress line animates height: 0 to 100%
// Timeline positions with negative overlap
// Items animate x: -20 (from left)
// Dots change color as timeline progresses
