JavaScript Animation
// Split each line into characters using custom splitText
const split1 = splitText("#line1", "chars");
const split2 = splitText("#line2", "chars");
const split3 = splitText("#line3", "chars");

// Set initial state
[split1, split2, split3].forEach(split => {
  gsap.set(split, { opacity: 0, y: 80 });
});

// Create master timeline
const tl = gsap.timeline();

// Line 1 - 0.5s duration
tl.to(split1, {
  opacity: 1,
  y: 0,
  stagger: 0.04,
  duration: 0.5,
  ease: "power3.out"
})

// Line 2 - starts 0.2s before line 1 ends (overlap)
.to(split2, {
  opacity: 1,
  y: 0,
  stagger: 0.04,
  duration: 0.5,
  ease: "power3.out"
}, "-=0.25")

// Line 3
.to(split3, {
  opacity: 1,
  y: 0,
  stagger: 0.04,
  duration: 0.5,
  ease: "power3.out"
}, "-=0.25")

// Subline fades in last
.to("#subline", {
  opacity: 1,
  duration: 0.8,
  ease: "power2.out"
}, "+=0.2");
Key Concepts
// Timeline positioning with overlap:
// "-=0.25" means start 0.25s before previous ends
// "+=0.2" means start 0.2s after previous ends

// power3.out gives more "punch" than power2.out
// Stagger of 0.04 = 40ms between characters
