JavaScript Animation
// Target price value
const targetPrice = 299;

// Create counter animation
const counter = { value: 0 };

const tl = gsap.timeline();

// Animate the number from 0 to 299
tl.to(counter, {
  value: targetPrice,
  duration: 1.5,
  ease: "power2.out",
  onUpdate: function() {
    // Update the display on each frame
    document.getElementById("price-value").textContent = 
      Math.round(counter.value);
  }
})
.fromTo(".price-period",
  { opacity: 0, y: 10 },
  { opacity: 1, y: 0, duration: 0.5 }
)
.fromTo(".price-feature",
  { opacity: 0, y: 20 },
  { opacity: 1, y: 0, stagger: 0.1, duration: 0.5 },
  "-=0.3"
);
Key Concepts
// Object-based animation for numbers
// counter = { value: 0 } - proxy object
// onUpdate fires on every animation frame
// Math.round() ensures clean integer display
