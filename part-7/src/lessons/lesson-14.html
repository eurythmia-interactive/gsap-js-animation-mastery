JavaScript Animation
// Get all bars
const bars = gsap.utils.toArray(".bar");

// Set initial state - bars at 0 height
bars.forEach(bar => {
  gsap.set(bar, { height: 0 });
  bar.querySelector(".bar-value").style.opacity = 0;
});

// Create staggered animation
const tl = gsap.timeline();

tl
  // Bars grow up
  .to(bars, {
    height: (i) => bars[i].dataset.height + "%",
    stagger: 0.15,
    duration: 0.8,
    ease: "power2.out"
  })
  // Labels fade in
  .fromTo(".bar-label",
    { opacity: 0, y: 10 },
    { opacity: 1, y: 0, stagger: 0.1, duration: 0.4 },
    "-=0.6"
  )
  // Values pop in at top of bars
  .to(".bar-value",
    { opacity: 1, stagger: 0.1, duration: 0.3, ease: "back.out(2)" },
    "-=0.4"
  );
Key Concepts
// data-height attribute stores target height
// height: (i) => dynamic value per element
// Negative position in timeline "-=0.6"
// Creates synced chart + label reveal
