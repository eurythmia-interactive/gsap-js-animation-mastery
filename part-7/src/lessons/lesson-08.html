JavaScript Animation
// Select all service cards
const cards = gsap.utils.toArray(".service-card");

// Set initial state - all invisible and below
gsap.set(cards, {
  opacity: 0,
  y: 60,
  scale: 0.9
});

// Animate cards with stagger
const tl = gsap.timeline();

tl.to(cards, {
  opacity: 1,
  y: 0,
  scale: 1,
  stagger: {
    amount: 0.8,  // Total stagger time
    from: "start"
  },
  duration: 0.8,
  ease: "back.out(1.4)"
});

// Alternative: stagger by grid position
// stagger: { each: 0.1, grid: [2, 3], from: "start" }
Key Concepts
// gsap.utils.toArray converts selector to array
// stagger: { amount: 0.8 } = spread over 800ms total
// stagger: { grid: [2, 3] } = 2 rows, 3 columns
// back.out(1.4) = slight overshoot for life
