HTML Structure
<div class="progress-container">
  <div class="progress-header">
    <span class="progress-label">Quiz Completion</span>
    <span class="progress-percent" id="progress-percent">0%</span>

  <div class="progress-bar-bg">
    <div class="progress-bar-fill" id="progress-fill"></div>
  </div>
</div>
JavaScript Animation
const targetPercent = 60;  // 3 out of 5 = 60%
const counter = { value: 0 };

const tl = gsap.timeline();

// Animate percentage number
tl.to(counter, {
  value: targetPercent,
  duration: 1.5,
  ease: "power2.out",
  onUpdate: () => {
    document.getElementById("progress-percent").textContent = 
      Math.round(counter.value) + "%";
  }
})

// Animate progress bar width
.to("#progress-fill", {
  width: targetPercent + "%",
  duration: 1.5,
  ease: "power2.out"
}, "<")  // "<" = same start time as previous

// Pulse effect on completion
.to("#progress-percent", {
  scale: 1.2,
  duration: 0.3,
  ease: "back.out(2)"
})
.to("#progress-percent", {
  scale: 1,
  duration: 0.3,
  ease: "elastic.out(1, 0.5)"
});
Key Concepts
// "<" in timeline = start at same time as previous
// Counter object for animated percentage
// onUpdate fires on every frame during animation
// Pulse effect adds satisfaction to completion
