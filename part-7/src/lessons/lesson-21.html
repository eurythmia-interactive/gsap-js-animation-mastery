JavaScript Animation
// Split title for dramatic reveal using custom splitText
const titleSplit = splitText("#credits-title", "chars");
const credits = gsap.utils.toArray(".credits-item");

// Set initial states
gsap.set(titleSplit, { opacity: 0, y: 50, scale: 0.5 });
gsap.set("#credits-subtitle, .credits-divider, .credits-footer", { opacity: 0 });
gsap.set(credits, { opacity: 0, y: 30 });
gsap.set(".glow-ring", { opacity: 0, scale: 0.5 });

const tl = gsap.timeline();

tl
  // Glow rings expand outward
  .to(".glow-ring", {
    opacity: 0.5,
    scale: 1,
    stagger: 0.3,
    duration: 1.5,
    ease: "power2.out"
  })
  // Title characters reveal with scale
  .to(titleSplit, {
    opacity: 1,
    y: 0,
    scale: 1,
    stagger: 0.08,
    duration: 0.8,
    ease: "back.out(1.5)"
  }, "-=1")
  // Subtitle fades in
  .to("#credits-subtitle", {
    opacity: 1,
    duration: 1,
    ease: "power2.out"
  }, "-=0.3")
  // Divider draws
  .fromTo("#credits-divider",
    { scaleX: 0 },
    { opacity: 1, scaleX: 1, duration: 1, ease: "power2.inOut" }
  , "+=0.3")
  // Credits items appear one by one
  .to(credits, {
    opacity: 1,
    y: 0,
    stagger: 0.8,
    duration: 0.8,
    ease: "power2.out"
  }, "-=0.5")
  // Final thank you message
  .to("#credits-footer", {
    opacity: 1,
    duration: 1.5,
    ease: "power2.out"
  }, "+=0.5");
Key Concepts
// Multiple elements = orchestrated finale
// Ring expansion creates depth
// back.out for playful character reveal
// Timeline sequences for dramatic ending
