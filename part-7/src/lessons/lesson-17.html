JavaScript Animation
// Get all words
const words = gsap.utils.toArray(".word");

// Set initial state - invisible
gsap.set(words, { opacity: 0, y: 20 });

// Create timeline with word-by-word reveal
const tl = gsap.timeline();

tl.to(words, {
  opacity: 1,
  y: 0,
  stagger: 0.2,  // 200ms between words
  duration: 0.5,
  ease: "power2.out"
})
// Highlighted words get extra animation
.to(".highlight-word", {
  scale: 1.2,
  duration: 0.3,
  ease: "back.out(3)"
}, "-=0.5")
// Add glow effect to highlights
.to(".highlight-word", {
  textShadow: "0 0 20px currentColor",
  duration: 0.5,
  ease: "power2.out"
}, "-=0.3");
Key Concepts
// Each word is its own element for animation
// .highlight-word class marks emphasis
// textShadow creates glow effect
// scale: 1.2 pops highlighted words out
