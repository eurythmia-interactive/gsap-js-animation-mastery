JavaScript with ScrollTrigger
// Register ScrollTrigger
gsap.registerPlugin(ScrollTrigger);

// Get all story sections
const sections = gsap.utils.toArray(".story-section");

// Animate each section on scroll
sections.forEach((section, index) => {
  // The chapter label and text animate together
  const chapter = section.querySelector(".story-chapter");
  const text = section.querySelector(".story-text");
  
  // Set initial state
  gsap.set(section, { opacity: 0, y: 60 });
  
  // Create scroll-triggered animation
  gsap.to(section, {
    opacity: 1,
    y: 0,
    duration: 1,
    ease: "power2.out",
    scrollTrigger: {
      trigger: section,
      start: "top 80%",
      end: "top 40%",
      scrub: 1 // Smooth scroll-linked animation
    }
  });
  
  // Highlighted words pulse on scroll
  const highlights = section.querySelectorAll(".story-highlight");
  highlights.forEach(hl => {
    gsap.to(hl, {
      scale: 1.05,
      duration: 0.5,
      ease: "power2.out",
      scrollTrigger: {
        trigger: hl,
        start: "top 80%",
        toggleActions: "play none none reverse"
      }
    });
  });
});
Key Concepts
// ScrollTrigger with scrub: 1
// Animation tied to scroll position
// Each section is independent trigger
// Highlights pulse on their own trigger
