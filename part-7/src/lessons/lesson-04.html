JavaScript Animation
// Split product name into words using custom splitText
const splitName = splitText("#product-name", "words");

// Initial state - scaled down and invisible
gsap.set(splitName, {
  opacity: 0,
  scale: 0.5,
  transformOrigin: "center center"
});

gsap.set("#product-tagline", { opacity: 0, y: 20 });

// Create timeline
const tl = gsap.timeline();

tl
  // Badge appears first
  .fromTo(".product-badge", 
    { opacity: 0, y: -20 },
    { opacity: 1, y: 0, duration: 0.5, ease: "back.out(1.7)" }
  )
  // Product name words scale in with stagger
  .to(splitName, {
    opacity: 1,
    scale: 1,
    stagger: 0.15,
    duration: 0.8,
    ease: "elastic.out(1, 0.5)"
  }, "-=0.2")
  // Tagline fades in
  .to("#product-tagline", {
    opacity: 1,
    y: 0,
    duration: 0.6,
    ease: "power2.out"
  }, "-=0.3");
Key Concepts
// elastic.out creates a bouncy, playful feel
// Perfect for e-commerce product reveals
// scale: 0.5 means element starts at half size
// transformOrigin ensures scaling from center
