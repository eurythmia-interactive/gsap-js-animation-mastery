JavaScript Animation
// Split title into words for dramatic reveal using custom splitText
const titleSplit = splitText("#chapter-title", "words");

// Set initial states
gsap.set(titleSplit, { opacity: 0, y: 80 });
gsap.set("#chapter-label, #chapter-subtitle, #chapter-number", { opacity: 0 });

// Cinematic timeline
const tl = gsap.timeline();

tl
  // Chapter label fades in first
  .to("#chapter-label", {
    opacity: 1,
    duration: 1.5,
    ease: "power2.out"
  })
  // Title words rise dramatically
  .to(titleSplit, {
    opacity: 1,
    y: 0,
    stagger: 0.25,
    duration: 1,
    ease: "power3.out"
  }, "-=0.5")
  // Subtitle fades in slowly
  .to("#chapter-subtitle", {
    opacity: 1,
    duration: 2,
    ease: "power2.out"
  }, "+=0.5")
  // Chapter number at the end
  .fromTo("#chapter-number",
    { opacity: 0, letterSpacing: "0.5em" },
    { opacity: 1, letterSpacing: "0.2em", duration: 1.5 },
    "-=1"
  );
Key Concepts
// Cinematic = slow, deliberate, atmospheric
// power3.out for dramatic weight
// Vignette overlay adds cinematic feel
// letterSpacing animates for elegance
