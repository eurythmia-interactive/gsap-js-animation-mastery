JavaScript Animation
// Get question text content
const questionEl = document.getElementById("question-text");
const text = questionEl.textContent;
questionEl.textContent = "";

// Split into individual characters
const chars = text.split("").map(char => {
  const span = document.createElementNS("http://www.w3.org/2000/svg", "span");
  span.textContent = char;
  span.style.display = "inline-block";
  span.style.opacity = "0";
  questionEl.appendChild(span);
  return span;
});

// Create timeline
const tl = gsap.timeline();

tl
  // Question number slides in
  .fromTo("#question-number",
    { opacity: 0, x: -20 },
    { opacity: 1, x: 0, duration: 0.4, ease: "back.out(1.5)" }
  )
  // Characters appear one by one (typewriter effect)
  .to(chars, {
    opacity: 1,
    stagger: 0.03,  // 30ms per character
    duration: 0.1,
    ease: "none"
  }, "-=0.2")
  // Progress dots animate
  .fromTo(".progress-dot",
    { scale: 0 },
    { scale: 1, stagger: 0.1, duration: 0.3, ease: "back.out(2)" },
    "-=0.5"
  );
Key Concepts
// Typewriter effect = stagger with very small delay
// stagger: 0.03 = 30ms between characters
// Creates anticipation and engagement
// Great for quiz questions, dialog, stories
