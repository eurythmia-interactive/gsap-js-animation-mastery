JavaScript Animation
function playAnimation() {
  const btn = document.getElementById("add-btn");
  const btnText = document.getElementById("btn-text");
  const cartIcon = document.getElementById("cart-icon");
  const cartCount = document.getElementById("cart-count");
  
  const tl = gsap.timeline();
  
  // Button press effect
  tl.to(btn, {
    scale: 0.95,
    duration: 0.1
  })
  // Button text changes
  .to(btnText, {
    opacity: 0,
    y: -10,
    duration: 0.15,
    onComplete: () => btnText.textContent = "Added!"
  })
  .to(btnText, {
    opacity: 1,
    y: 0,
    duration: 0.15
  })
  // Button returns to normal
  .to(btn, {
    scale: 1,
    duration: 0.1
  })
  // Cart icon bounces
  .to(cartIcon, {
    scale: 1.3,
    duration: 0.2,
    ease: "back.out(3)"
  })
  .to(cartIcon, {
    scale: 1,
    duration: 0.3,
    ease: "elastic.out(1, 0.5)"
  })
  // Cart count appears and increments
  .to(cartCount, {
    opacity: 1,
    scale: 1.2,
    duration: 0.2
  }, "-=0.2")
  .to(cartCount, {
    scale: 1,
    duration: 0.2
  });
}
Key Concepts
// onComplete callback for text changes
// elastic.out for bouncy cart effect
// back.out for satisfying overshoot
// Scale animations must reset properly
