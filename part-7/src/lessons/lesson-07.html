JavaScript Animation
// Split headline into words using custom splitText
const split = splitText("#headline", "words");

// Set initial state - all invisible
gsap.set(split, { opacity: 0, y: 40 });
gsap.set("#divider", { opacity: 0, scaleX: 0, transformOrigin: "left center" });
gsap.set("#tagline, #body", { opacity: 0, y: 20 });

// Create slow, dignified timeline
const tl = gsap.timeline();

tl
  // Headline words appear one by one
  .to(split, {
    opacity: 1,
    y: 0,
    stagger: 0.4,  // 400ms between words
    duration: 1,
    ease: "power2.out"
  })
  // Divider draws from left to right
  .to("#divider", {
    opacity: 1,
    scaleX: 1,
    duration: 0.8,
    ease: "power2.inOut"
  }, "+=0.3")
  // Tagline fades in
  .to("#tagline", {
    opacity: 1,
    y: 0,
    duration: 1,
    ease: "power2.out"
  }, "+=0.2")
  // Body text last
  .to("#body", {
    opacity: 1,
    y: 0,
    duration: 1,
    ease: "power2.out"
  }, "+=0.3");
Key Concepts
// Corporate = slow, deliberate timing
// stagger: 0.4 (400ms) = dignified pacing
// power2.out = professional, not flashy
// Line reveals build trust and credibility
