JavaScript Animation
// Get all answer options
const options = gsap.utils.toArray(".answer-option");

// Set initial state - all invisible, scaled down
gsap.set(options, {
  opacity: 0,
  scale: 0.8,
  y: 20
});

// Get answer letters for special animation
const letters = gsap.utils.toArray(".answer-letter");

// Create timeline
const tl = gsap.timeline();

tl
  // Question appears
  .fromTo("#question",
    { opacity: 0, y: -20 },
    { opacity: 1, y: 0, duration: 0.5, ease: "power2.out" }
  )
  // Options appear with stagger
  .to(options, {
    opacity: 1,
    scale: 1,
    y: 0,
    stagger: 0.15,  // 150ms between options
    duration: 0.5,
    ease: "back.out(1.5)"
  }, "-=0.2")
  // Letters have a pop effect
  .fromTo(letters,
    { scale: 0 },
    { scale: 1, stagger: 0.15, duration: 0.4, ease: "elastic.out(1, 0.5)" },
    "-=0.6"
  );
Key Concepts
// back.out creates slight overshoot
// elastic.out for bouncy letter circles
// stagger: 0.15 = good pacing for reading
// Nested animations within timeline
