JavaScript Animation
// Counter objects for animation
const counters = [
  { id: "stat-1", target: 24, suffix: "K" },
  { id: "stat-2", target: 97, suffix: "%" },
  { id: "stat-3", target: 156, suffix: "" }
].map(c => ({ ...c, value: 0 }));

const tl = gsap.timeline();

counters.forEach((counter, index) => {
  tl.to(counter, {
    value: counter.target,
    duration: 2,
    ease: "power2.out",
    onUpdate: () => {
      document.getElementById(counter.id).textContent = 
        Math.round(counter.value);
    },
    onComplete: () => {
      // Add suffix after counting completes
      const el = document.getElementById(counter.id);
      if (counter.suffix && !el.textContent.includes(counter.suffix)) {
        el.textContent = Math.round(counter.value) + counter.suffix;
      }
    }
  }, index * 0.2); // 200ms stagger
});
Key Concepts
// Map array to add 'value' property to each counter
// onComplete for post-animation modifications
// Stagger in timeline with index * 0.2
// Data viz = big numbers, bold fonts
