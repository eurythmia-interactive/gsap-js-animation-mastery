CSS Environment Variables

Key Code
<!-- viewport-fit=cover required -->
<meta name="viewport" content="width=device-width, 
     initial-scale=1.0, viewport-fit=cover">

<!-- Full-bleed gradient background -->
<div class="bg-gradient-to-b from-rose-500 to-orange-500 
               pt-[env(safe-area-inset-top)] 
               pb-[env(safe-area-inset-bottom)]">
  <!-- Content extends edge to edge -->
</div>

<!-- Inset content with safe margins -->
<main class="pt-[calc(env(safe-area-inset-top)+16px)] 
               pb-[calc(env(safe-area-inset-bottom)+16px)] 
               px-4">
  <!-- Safe content area -->
</main>

<!-- Tab bar with safe area -->
<nav class="pb-[env(safe-area-inset-bottom)] pt-4 
               bg-black">
  <!-- Navigation content -->
</nav>
Fallback Values
/* Fallback for older browsers */
padding-top: max(env(safe-area-inset-top), 20px);

/* Safe area with custom fallback */
padding-bottom: calc(
  env(safe-area-inset-bottom, 20px) + 16px
);
