Card Design Guidelines
/* Mobile card corner radius */
rounded-lg:  8px;   /* Small elements */
rounded-xl: 12px;  /* Buttons, inputs */
rounded-2xl: 16px; /* Cards */
rounded-3xl: 24px; /* Large cards */

/* Card padding scale */
p-4:  16px;  /* Compact */
p-5:  20px;  /* Standard */
p-6:  24px;  /* Spacious */
